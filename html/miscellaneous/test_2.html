shots_count = df.groupby(['shots_fired']).count()

shots_count = shots_count.drop(['agency','city','county','incident_date',
                                'officer_names','outcome','searched_date',
                                'source_url','state','summary','victim_age',
                                'victim_armed','victim_gender','victim_race',
                                'victim_name','weapon'], axis=1)

shots_count = shots_count.rename(columns={"shootings": "count"})

shots_count['shots'] = shots_count.index
shots_count['shots'] = shots_count['shots'].astype(int)

shots_count['percent'] = shots_count['count'] / shots_count['count'].sum() * 100
shots_count['percent'] = shots_count['percent'].round(decimals=1)
