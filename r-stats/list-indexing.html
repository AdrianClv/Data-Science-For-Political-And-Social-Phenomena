# create a list with simulated values
score <- runif(100)
states.df <- data.frame(state1 = state.name[1:10], state2 = state.name[11:20], state3 = state.name[21:30])
name <- letters[1:20]
data.ls <- list(score, states.df, name)
rm(score, states.df, name)
