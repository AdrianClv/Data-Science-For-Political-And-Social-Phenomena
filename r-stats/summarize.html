Summarize A Variable

Thu 01 December 2016 
# Create two variables of 50 observations
percent.sms <- runif(50)
location <- state.name

# Create a dataframe of those two variables
usa <- data.frame(location, percent.sms)

# Summarize the dataframe
summary(usa)
