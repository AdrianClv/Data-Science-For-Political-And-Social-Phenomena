# create a matrix of simulated count data
counts <- matrix(sample(1:100, 20, replace=T), nrow=4); counts

     [,1] [,2] [,3] [,4] [,5]
[1,]   77   79   14   94    7
[2,]  100   54   86   28   60
[3,]   38   99   77   84   81
[4,]   75   93   19   21   39

prop.table(counts,1)
